CREATE TABLE IF NOT EXISTS public.cliente
(
    nome character varying(100) COLLATE pg_catalog."default",
    numero_polizza character varying(50) COLLATE pg_catalog."default" NOT NULL,
    scadenza date,
    premio character varying(255) COLLATE pg_catalog."default",
    parcella character varying(255) COLLATE pg_catalog."default",
    agenzia character varying(100) COLLATE pg_catalog."default",
    pagato boolean DEFAULT false,
    frazionamento character(1) COLLATE pg_catalog."default",
    data_pagamento date,
    fattura character varying(255) COLLATE pg_catalog."default",
    indicizzazione boolean,
    CONSTRAINT cliente_pkey PRIMARY KEY (numero_polizza),
    CONSTRAINT cliente_frazionamento_check CHECK (frazionamento = ANY (ARRAY['A'::bpchar, 'S'::bpchar]))
)
TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.cliente
    OWNER to postgres;
